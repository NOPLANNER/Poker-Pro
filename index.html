function calcLogic() { if(!usedCards['h1'] || !usedCards['h2']) return; const [s1, r1] = usedCards['h1'].split('-'); const [s2, r2] = usedCards['h2'].split('-'); const v1 = 14 - ranks.indexOf(r1), v2 = 14 - ranks.indexOf(r2); const char1 = (r1 === '10' ? 'T' : r1), char2 = (r2 === '10' ? 'T' : r2); const handStr = (v1===v2) ? char1+char1 : (v1>v2 ? char1+char2 : char2+char1) + (s1===s2 ? 's' : 'o'); // Equity 簡易 let eq = (v1 + v2) * 1.5 + (v1 === v2 ? 20 : 0) + (s1 === s2 ? 5 : 0); document.getElementById('eq-val').innerText = Math.min(Math.max(eq, 2.0), 99.9).toFixed(1) + "%"; // --- ロジック修正: HeroからBBまでの人数を正確にカウント --- const heroPosIdx = (0 - btnPosition + playerCount) % playerCount; let countToBB; if (heroPosIdx === 0) { countToBB = 3; // BTN, SB, BB の3人 } else if (heroPosIdx === 1) { countToBB = 2; // SB, BB の2人 } else if (heroPosIdx === 2) { countToBB = 1; // BB のみ1人 } else { // UTGやCOなど：(自分からBTNまでの人数) + (SB, BBの2人) countToBB = (playerCount - heroPosIdx) + 2; } // ホワイトレンジの「2人以下」は SB, BB の時のみ。 // ボタン(3人)でオープンしたい場合は、ホワイトの limit を 3 に調整します。 // ご提示の条件「BBまで2人以下」にボタンを含めるよう調整しました。 let resultAction = "FOLD", resultColor = "#888"; for (const key in yokosawa_rules) { const rule = yokosawa_rules[key]; if (rule.hands.includes(handStr)) { // ホワイト(white)かつBTN(3)の場合にオープンさせるための特例、またはlimitの解釈修正 let effectiveLimit = rule.limit; if (key === 'white' && effectiveLimit === 2) effectiveLimit = 3; if (key === 'cyan' && effectiveLimit === 3) effectiveLimit = 4; // 必要に応じてスライド // もしくは、シンプルに「BTN＝3人、SB＝2人、BB＝1人」として // ホワイトの条件を 3人以下（BTNまで）にするのが一般的です if (countToBB <= (rule.limit === 2 ? 3 : rule.limit)) { resultAction = "OPEN"; resultColor = rule.color; } break; } } document.getElementById('yokosawa-text').innerText = resultAction; document.getElementById('yokosawa-text').style.color = resultColor; document.getElementById('gto-text').innerText = (eq > 50) ? "RAISE" : "FOLD"; }
